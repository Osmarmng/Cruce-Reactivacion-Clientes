""" # ── Lógica del cruce ──────────────────────────────────────────────────────── def parse_date(val): if val is None: return None if isinstance(val, datetime): return val try: return pd.to_datetime(str(val), dayfirst=True) except: return None def fmt_date(val): d = parse_date(val) if d and not pd.isna(d): return d.strftime('%d/%m/%Y') return str(val)[:10] if val else '' COLOR_IGNORAR = {'FF92D050', 'FFFF0000', 'FFFFC000', 'FF00B050', 'FFFF0000'} def get_color(cell): try: fill = cell.fill if fill and fill.fgColor: fc = fill.fgColor if fc.type == 'rgb': return fc.rgb.upper() except: pass return '00000000' def procesar_cruce(cc_bytes, susp_bytes): today = datetime.today().replace(hour=0, minute=0, second=0, microsecond=0) # ── Leer cuenta corriente ── import xlrd try: cc_wb = openpyxl.load_workbook(io.BytesIO(cc_bytes), read_only=True, data_only=True) cc_sheet_name = next((s for s in cc_wb.sheetnames if 'cuenta corriente' in s.lower()), cc_wb.sheetnames[0]) cc_ws = cc_wb[cc_sheet_name] cc_rows = list(cc_ws.iter_rows(values_only=True)) except: # Fallback para .xls legacy cc_df = pd.read_excel(io.BytesIO(cc_bytes), engine='xlrd') cc_rows = [tuple(cc_df.columns)] + [tuple(r) for r in cc_df.values] # Detectar columnas header = [str(c).strip() if c else '' for c in cc_rows[0]] def find_col(keywords): for kw in keywords: for i, h in enumerate(header): if kw.lower() in h.lower(): return i return None rut_idx = find_col(['RUT', 'rut']) fv_idx = find_col(['FechaVencimiento', 'Fecha Vencimiento', 'vencimiento']) saldo_idx = find_col(['Saldo', 'saldo']) if rut_idx is None or fv_idx is None or saldo_idx is None: return {'error': f'No se encontraron columnas requeridas en cuenta corriente. Columnas: {header}'} vencidas = {} for row in cc_rows[1:]: try: rut = str(row[rut_idx]).strip() if row[rut_idx] else '' saldo = float(row[saldo_idx] or 0) fv = parse_date(row[fv_idx]) if rut and fv and fv < today and saldo > 0: if rut not in vencidas: vencidas[rut] = {'count': 0, 'total': 0.0, 'oldest': fv} vencidas[rut]['count'] += 1 vencidas[rut]['total'] += saldo if fv < vencidas[rut]['oldest']: vencidas[rut]['oldest'] = fv except: continue # ── Leer suspendidos (hoja BASE, solo sin color) ── susp_wb = openpyxl.load_workbook(io.BytesIO(susp_bytes), data_only=True) base_sheet = next((s for s in susp_wb.sheetnames if s.upper() == 'BASE'), susp_wb.sheetnames[0]) susp_ws = susp_wb[base_sheet] rows = list(susp_ws.iter_rows()) if not rows: return {'error': 'La hoja BASE está vacía'} # Detectar columnas en suspendidos hdr_susp = [str(c.value).strip() if c.value else '' for c in rows[0]] def find_s(kws): for kw in kws: for i, h in enumerate(hdr_susp): if kw.lower() in h.lower(): return i return None s_rut = find_s(['rut']) s_nombre = find_s(['razon_social','razon social','nombre','cliente']) s_corte = find_s(['corte','fecha']) total = 0 ignorados = 0 data_ok = [] data_nok = [] for row in rows[1:]: if not any(c.value for c in row): continue total += 1 color = get_color(row[0]) if color not in ('00000000', 'FFFFFFFF', ''): ignorados += 1 continue rut = str(row[s_rut].value).strip() if s_rut is not None and row[s_rut].value else '' nombre = str(row[s_nombre].value).strip() if s_nombre is not None and row[s_nombre].value else '' corte = fmt_date(row[s_corte].value) if s_corte is not None else '' deuda = vencidas.get(rut) if not deuda: data_ok.append({'rut': rut, 'nombre': nombre, 'corte': corte}) else: data_nok.append({ 'rut': rut, 'nombre': nombre, 'corte': corte, 'facturas': deuda['count'], 'saldo': round(deuda['total']), 'oldest': fmt_date(deuda['oldest']) }) return { 'ok': data_ok, 'nok': data_nok, 'total': total, 'ignorados': ignorados, 'sincolor': total - ignorados } def generar_excel(ok_rows, nok_rows): from openpyxl import Workbook from openpyxl.styles import Font, PatternFill, Alignment, Border, Side wb = Workbook() thin = Side(style='thin') def brd(c): c.border = Border(left=thin,right=thin,top=thin,bottom=thin) def hdr(ws, row, cols, bg): for i, h in enumerate(cols, 1): c = ws.cell(row=row, column=i, value=h) c.font = Font(bold=True, color='FFFFFF', name='Arial', size=10) c.fill = PatternFill('solid', start_color=bg) c.alignment = Alignment(horizontal='center', vertical='center') brd(c) # Hoja 1 ws1 = wb.active; ws1.title = 'Candidatos a Reactivar' ws1.merge_cells('A1:D1') ws1['A1'] = f'Candidatos a Reactivar — {datetime.today().strftime("%d/%m/%Y")}' ws1['A1'].font = Font(bold=True, color='FFFFFF', name='Arial', size=11) ws1['A1'].fill = PatternFill('solid', start_color='375623') ws1['A1'].alignment = Alignment(horizontal='center', vertical='center') ws1.row_dimensions[1].height = 24 hdr(ws1, 2, ['RUT','Razón Social','Fecha Corte','Estado'], '375623') for r, row in enumerate(ok_rows, 3): vals = [row['rut'], row['nombre'], row['corte'], 'Al día'] for i, v in enumerate(vals, 1): c = ws1.cell(row=r, column=i, value=v) c.fill = PatternFill('solid', start_color='C6EFCE') c.font = Font(name='Arial', size=10) brd(c) ws1.column_dimensions['A'].width = 16; ws1.column_dimensions['B'].width = 48 ws1.column_dimensions['C'].width = 16; ws1.column_dimensions['D'].width = 12 # Hoja 2 ws2 = wb.create_sheet('Con Deuda Vencida') ws2.merge_cells('A1:F1') ws2['A1'] = f'Con Deuda Vencida — {datetime.today().strftime("%d/%m/%Y")}' ws2['A1'].font = Font(bold=True, color='FFFFFF', name='Arial', size=11) ws2['A1'].fill = PatternFill('solid', start_color='9C0006') ws2['A1'].alignment = Alignment(horizontal='center', vertical='center') ws2.row_dimensions[1].height = 24 hdr(ws2, 2, ['RUT','Razón Social','Fecha Corte','Fact. Vencidas','Saldo ($)','Venc. más Antiguo'], '9C0006') for r, row in enumerate(nok_rows, 3): vals = [row['rut'], row['nombre'], row['corte'], row['facturas'], row['saldo'], row['oldest']] for i, v in enumerate(vals, 1): c = ws2.cell(row=r, column=i, value=v) c.fill = PatternFill('solid', start_color='FFCCCC') c.font = Font(name='Arial', size=10) brd(c) ws2.cell(row=r, column=5).number_format = '#,##0' for col, w in zip('ABCDEF', [16,48,16,18,16,20]): ws2.column_dimensions[col].width = w # Hoja 3 Resumen ws3 = wb.create_sheet('Resumen') ws3.merge_cells('A1:B1') ws3['A1'] = f'Resumen — {datetime.today().strftime("%d/%m/%Y")}' ws3['A1'].font = Font(bold=True, color='FFFFFF', name='Arial', size=11) ws3['A1'].fill = PatternFill('solid', start_color='1F4E79') ws3['A1'].alignment = Alignment(horizontal='center') for r, (lbl, val) in enumerate([ ('Candidatos a reactivar (sin deuda vencida)', len(ok_rows)), ('Con deuda vencida pendiente', len(nok_rows)), ('Total procesados', len(ok_rows)+len(nok_rows)), ], 3): ws3.cell(row=r, column=1, value=lbl).font = Font(name='Arial', size=10) ws3.cell(row=r, column=2, value=val).font = Font(name='Arial', size=10, bold=True) ws3.column_dimensions['A'].width = 48; ws3.column_dimensions['B'].width = 14 buf = io.BytesIO() wb.save(buf) return buf.getvalue() # ── Servidor HTTP ─────────────────────────────────────────────────────────── class Handler(BaseHTTPRequestHandler): def log_message(self, fmt, *args): pass # silenciar logs def do_GET(self): self.send_response(200) self.send_header('Content-Type', 'text/html; charset=utf-8') self.end_headers() self.wfile.write(HTML.encode('utf-8')) def do_POST(self): length = int(self.headers.get('Content-Length', 0)) body = self.rfile.read(length) data = json.loads(body) if self.path == '/cruce': try: cc_bytes = base64.b64decode(data['cc']) susp_bytes = base64.b64decode(data['susp']) result = procesar_cruce(cc_bytes, susp_bytes) self.send_response(200) self.send_header('Content-Type', 'application/json') self.end_headers() self.wfile.write(json.dumps(result).encode('utf-8')) except Exception as e: self.send_response(500) self.send_header('Content-Type', 'application/json') self.end_headers() self.wfile.write(json.dumps({'error': str(e)}).encode()) elif self.path == '/export': try: xlsx = generar_excel(data['ok'], data['nok']) self.send_response(200) self.send_header('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet') self.send_header('Content-Disposition', 'attachment; filename="cruce.xlsx"') self.end_headers() self.wfile.write(xlsx) except Exception as e: self.send_response(500) self.end_headers() # ── Main ──────────────────────────────────────────────────────────────────── if __name__ == '__main__': PORT = 8765 server = HTTPServer(('localhost', PORT), Handler) url = f'http://localhost:{PORT}' print(f'\n✅ Panel iniciado en {url}') print(' Abriendo navegador...') print(' Para cerrar el panel, cierra esta ventana o presiona Ctrl+C\n') threading.Timer(1.2, lambda: webbrowser.open(url)).start() try: server.serve_forever() except KeyboardInterrupt: print('\nPanel cerrado.')
